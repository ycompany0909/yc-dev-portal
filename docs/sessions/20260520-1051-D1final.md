```js
function readEmail(req) {
  const hdr = (req.headers.get('Cf-Access-Authenticated-User-Email') || '').toLowerCase().trim();
  if (hdr) return hdr;
  const jwt = req.headers.get('Cf-Access-Jwt-Assertion');
  if (!jwt) return '';
  try {
    const p = jwt.split('.')[1];
    return (JSON.parse(atob(p.replace(/-/g, '+').replace(/_/g, '/'))).email || '').toLowerCase();
  } catch { return ''; }
}
```
